class Person<T>() {
	func sound() {}
}

class A extends Person {
  static val k: Int = 99

  constructor() {
    super()
  }

  func sound() {
	println("Hello A")
  }
}

class B extends Person {
  static val k: Int = 99

  constructor() {
    super()
  }

  func sound() {
	println("Hello B")
  }
}

func t(person: Person): Any {
	if (person is A) {
		return true
	}
	if (person is B) {
		return null
	}
	return 0
}

println(t(A() as? Person))
println(t(B()))
println(t((B() as Any) as? A))
