class T {
    static val c: Int = 1

    var name: String = "hi"

    func printName() {
        println(name)
    }

    func hm() {
        println("ok")
    }
}

class Person extends T {
    constructor() {
        super()
        println(name)
    }
}

class X extends Person {
    static val k: Int = 99
    val n: Int = 5

    constructor() {
        super()
    }

    func hm() {
        println(name)
        println(n)
        printName()
        println(c)
        println(k)
    }
}

val t = T()
t.hm()

val x = X()
x.hm()
